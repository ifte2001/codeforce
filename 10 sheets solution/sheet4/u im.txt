#include <stdio.h>
#include <string.h>

int main()
{
    char ch[1000010];     /// user string
    char a[10]="egypt";  ///we have to count how many character of "egypt" are there...
    int f[150]={0};      /// array for frequency of 'ch' string...

    printf("");  /// take input for ch...
    gets(ch);

    strlwr(ch);  /// there are upper and lowercase both in 'ch' string ... so we have to make all of them lower... it will make our work more easy...

    int n = strlen(ch);  /// length of ch string...

    for(int i=0;i<n;i++)  /// check ch string frequency...
    {
        int x = ch[i];
        f[x]++;
    }

    int num=0;  ///the number of character in "egypt" string...
    int flag = 1;  /// it will ON while loop... it will be off when the frequency of e,g,y,p,t will finish...
    while(flag==1)
    {
        int n_e = a[0]; ///int value for this character(ascii)...
        int n_g = a[1]; ///int value for this character(ascii)...
        int n_y = a[2]; ///int value for this character(ascii)...
        int n_p = a[3]; ///int value for this character(ascii)...
        int n_t = a[4]; ///int value for this character(ascii)...

        if(f[n_e]>=1 && f[n_g]>=1 && f[n_y]>=1 && f[n_p]>=1 && f[n_t]>=1)  ///we will decrease the frequency of e,g,y,p,t... untill they are greater then or equal one...
        {
            f[n_e]--;
            f[n_g]--;
            f[n_y]--;
            f[n_p]--;
            f[n_t]--;
            num++;   /// we will increase the value of num... untill this condition is true...
        }
        else   ///if any frequency of e,g,y,p,t will 0 ... then we will OFF the loop... by flag=0...
            flag=0;
    }

    printf("%d",num);  /// print the number of character in "egypt" string...

    return 0;
}
