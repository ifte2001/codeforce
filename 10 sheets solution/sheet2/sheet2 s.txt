#include<stdio.h>
int main()
{
    int n,a,b,sum,box;
    
    scanf("%d",&n);
    
    
    for(int i=1;i<=n;i+=1)
    {
      scanf("%d %d",&a,&b);
      
      if(a>b)
      {
        box = b;
        b = a;
        a = box;
      }
      
      sum=0;
      for(int j=a+1;j<b;j+=1)
      {
        if(j%2!=0)
        {
          sum = sum + j;
        }
      }
      printf("%d\n",sum);
    }
    
    returnÂ 0;
}
