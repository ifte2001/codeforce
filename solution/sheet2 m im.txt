#include<stdio.h>
int main()
{
    int n,i,j,l,a,b,x;

    scanf("%d %d",&a,&b);
    j = 0;

    for(int i=a;i<=b;i+=1)
    {
      n = i;
      x = 0;
      while(1)
      {
        l = n%10;
        n = n/10;

        if(l==4)
            x = 1;
        else if(l==7)
            x = 1;
        else
        {
            x = 0;
            break;
        }

        if(n==0)
            break;
      }

      if(x==0)
        continue;
      else
      {
        printf("%d ",i);
        j+=1;
      }
    }

    if(j==0)
        printf("-1\n");

    return 0;
}
