#include <stdio.h>

int main()
{
    int c,n,flag;

    scanf("%d",&c);

    for(int k=0;k<c;k++)
    {
        scanf("%d",&n);

        char a[n+1],b[n+1];

        scanf("%s",&a);
        scanf("%s",&b);

        for(int i=0;i<n;i++)
        {
            if(a[i]=='G' && b[i]=='B' || a[i]=='B' && b[i]=='G' || a[i]==b[i])  ///she will not understand blue and green... she will see those colors same...
                flag=1;   ///if she can't not understand the differents between two colors... then ok (means: 1)
            else
            {
                flag=0;   ///if she can understand the differents between two colors... then not ok (means: 0)
                break;
            }
        }

        if(flag==1)
            printf("YES\n");
        else if(flag==0)
            printf("NO\n");
    }
    return 0;
}
